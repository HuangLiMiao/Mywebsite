    <text font-family="microsoft yahei" font-size="120" y="150" x="160" text-color:"#FF6A88">
        欢迎你的到来
        <animate attributeName="x" values="160;40;160" dur="3s" repeatCount="indefinite" />
    </text>
</svg>
